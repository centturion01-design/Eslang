Entrada: "crea una carpeta llamada backups y copia todos los .txt allí"
Salida:
import os
import shutil

if __name__ == "__main__":
    os.makedirs("backups", exist_ok=True)
    copiados = 0
    for archivo in os.listdir("."):
        if archivo.endswith(".txt"):
            shutil.copy(archivo, f"backups/{archivo}")
            copiados += 1
    print(f"Se copiaron {copiados} archivos .txt a backups/")
